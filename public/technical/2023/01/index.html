#!/bin/sh

directory=$(find ~/myworkspace ~/Desktop -mindepth 0 -maxdepth 1 -type d | fzf)

if [ -z "$1" ]; then
    cd "$directory"
    exec bash
else
    exec ranger "$directory"
fi
